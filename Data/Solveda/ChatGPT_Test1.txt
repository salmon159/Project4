# Parameters

    month = i + 1
    year = i // 12 + 1
    future_value[i] = future_value[i - 1] * (1 + annual_return_rate / 12) if i > 0 else 0
    future_value[i] += monthly_sip

    if year >= start_withdrawal_year:
        withdrawal_year = year - start_withdrawal_year
